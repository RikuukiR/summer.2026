\subsection{最上級を使った構文(1)}
    \begin{theorembox}{\textbf{the＋最上級}}
        \begin{itemize}
            \item the ＋ 最上級 ＋ of ＋ (\fitblankbf[]{数を表す語句})
            \item the ＋ 最上級 ＋ in ＋ (\fitblankbf[]{場所・範囲を表す語句})
        \end{itemize}

        \begin{flushright}
            『～の中で最も…』
        \end{flushright}
    \end{theorembox}
    \vspace{0.5em}

\subsection{最上級を使った構文(2)}
    \begin{theorembox}{\textbf{one of the＋最上級}}
        one of the ＋ 最上級 ＋ (\fitblankbf[]{名詞の複数形})

        \begin{flushright}
            『\fitblankbf[]{最も}～な…の１つ』
        \end{flushright}
    \end{theorembox}
    \vspace{0.5em}

\subsection{最上級を使った構文(3)}
    \begin{theorembox}{\textbf{the＋序数＋最上級}}
        the ＋ (\fitblankbf[]{序数}) ＋ 最上級

        \begin{flushright}
            『～\fitblankbf[]{番目}に…な』
        \end{flushright}
    \end{theorembox}
    \vspace{0.5em}

    ※序数とは first, second, third などの数のことである。序数の\fitblankbf[]{位置}にも注意が必要である。

\subsection{練習問題}
    \begin{enumerate}
        \item 彼は日本で最も有名な歌手の1人だ。
        \dialogueblank{He is one of the most famous singers in Japan.}
        \vspace{0.5em}
        \item Mount Everest is the highest mountain in the world.
        \dialogueblank{エベレスト山は世界で一番高い山だ。}
        \vspace{0.5em}
        \item これは3つのうちで一番安い時計だ。
        \dialogueblank{This is the cheapest of the three watches.}
        \vspace{0.5em}
        \item She is the second tallest student in our class.
        \dialogueblank{彼女は私たちのクラスで2番目に背が高い生徒だ。}
        \vspace{0.5em}
        \item 札幌は日本で5番目に大きな都市だ。
        \dialogueblank{Sapporo is the fifth largest city in Japan.}
        \vspace{0.5em}
        \item This is one of the best movies I have ever seen.
        \dialogueblank{これは今まで見た中で最高の映画の1つだ。}
        \vspace{0.5em}
        \item 彼はクラスの中で最も上手に英語を話す。
        \dialogueblank{He speaks English best in the class.}
    \end{enumerate}
