\subsection{比較の大前提}
    大前提として、比較(比較級・最上級・原級)するのは\fitblankbf[]{形容詞}または\fitblankbf[]{副詞}である。

\subsection{比較級のつくり方}
    比較級のつくり方は以下の\fitblankbf[]{3パターン}である。

    \begin{definitionbox}{\textbf{比較級のつくり方}}
        \begin{enumerate}
            \item 基本は語句の語尾に\fitblankbf[]{r}、\fitblankbf[]{er}をつける
            \begin{itemize}
                \item tall (\fitblankbf[]{taller})
                \item easy (\fitblankbf[]{easier})
                \item nice (\fitblankbf[]{nicer})
                \item hot (\fitblankbf[]{hotter})
            \end{itemize}
            \item 長い単語(母音が\fitblankbf[]{3つ以上})のときは単語の前に\fitblankbf[]{more}をつける

            ※前に more をつけると比較級変化をしていることと同義なので r や er はつけない
            \begin{itemize}
                \item careful (\fitblankbf[]{more careful})
                \item beautiful (\fitblankbf[]{more beautiful})
                \item delicious (\fitblankbf[]{more delicious})
                \item interesting (\fitblankbf[]{more interesting})
            \end{itemize}
            \item 不規則に変化するもの (暗記必須！！)
            \begin{itemize}
                \item good, well (良い、上手な～) (\fitblankbf[]{better})
                \item many, much (多い、たくさんの～) (\fitblankbf[]{more})
                \item bad, ill (悪い、すぐれない～) (\fitblankbf[]{worse})
                \item little (小さい、少し) (\fitblankbf[]{less})
            \end{itemize}
        \end{enumerate}
    \end{definitionbox}
    \vspace{0.5em}

\subsection{練習問題}
    \begin{enumerate}
        \item tall
        \dialogueblank{taller}
        \vspace{0.5em}
        \item beautiful
        \dialogueblank{more beautiful}
        \vspace{0.5em}
        \item good
        \dialogueblank{better}
        \vspace{0.5em}
        \item この本はあの本より面白い。
        \dialogueblank{This book is more interesting than that book.}
        \vspace{0.5em}
        \item She runs faster than me.
        \dialogueblank{彼女は私より速く走る。}
        \vspace{0.5em}
        \item bad $\longrightarrow$ 比較級
        \dialogueblank{worse}
        \vspace{0.5em}
        \item much $\longrightarrow$ 比較級
        \dialogueblank{more}
    \end{enumerate}
