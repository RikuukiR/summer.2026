\subsection{最上級のつくり方}
    最上級のつくり方は以下の\fitblankbf[]{3パターン}である。

    \begin{definitionbox}{\textbf{最上級のつくり方}}
        \begin{enumerate}
            \item 基本は語句の語尾に\fitblankbf[]{st}、\fitblankbf[]{est}をつける
            \begin{itemize}
                \item poor (\fitblankbf[]{poorest})
                \item happy (\fitblankbf[]{happiest})
                \item simple (\fitblankbf[]{simplest})
                \item old (\fitblankbf[]{oldest})
                \item tall (\fitblankbf[]{tallest})
                \item big (\fitblankbf[]{biggest})
            \end{itemize}
            \item 長い単語(母音が\fitblankbf[]{3つ以上})のときは単語の前に\fitblankbf[]{most}をつける

            ※前に most をつけると最上級変化をしていることと同義なので st や est はつけない
            \begin{itemize}
                \item popular (\fitblankbf[]{most popular})
                \item difficult (\fitblankbf[]{most difficult})
            \end{itemize}
            \item 不規則に変化するもの (暗記必須！！)
            \begin{itemize}
                \item good, well (良い、上手な～) (\fitblankbf[]{best})
                \item many, much (多い、たくさんの～) (\fitblankbf[]{most})
                \item bad, ill (悪い、すぐれない～) (\fitblankbf[]{worst})
                \item little (小さい、少し) (\fitblankbf[]{least})
            \end{itemize}
        \end{enumerate}
    \end{definitionbox}
    \vspace{0.5em}
