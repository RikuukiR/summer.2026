\subsection{比較級を使った構文(1)}
    \begin{theorembox}{\textbf{Ａ＋比較級＋than＋Ｂ}}
        『\fitblankbf[]{B}よりも\fitblankbf[]{A}』
    \end{theorembox}
    \vspace{0.5em}

\subsection{比較級を使った構文(2)}
    比較級を強める語として\fitblankbf[]{much}、\fitblankbf[]{by far}、\fitblankbf[]{far}がある。
    これらはいずれも『\fitblankbf[]{ずっと}、\fitblankbf[]{はるかに}』と訳し、比較級の\fitblankbf[]{直前}に置く。

    \begin{techniquebox}{\textbf{比較級を強める語}}
        \begin{itemize}
            \item This box is \fitblankbf[]{much} heavier than that one.
            \item She is \fitblankbf[]{by far} taller than her sister.
            \item He runs \fitblankbf[]{far} faster than me.
        \end{itemize}
    \end{techniquebox}
    \vspace{0.5em}

\subsection{比較級を使った構文(3)}
    \begin{theorembox}{\textbf{(the)＋比較級＋of the two}}
        『\fitblankbf[]{2つのうち}より～な方』
    \end{theorembox}
    \vspace{0.5em}

\subsection{練習問題}
    \begin{enumerate}
        \item 彼は私より背が高い。
        \dialogueblank{He is taller than me.}
        \vspace{0.5em}
        \item My bag is much heavier than yours.
        \dialogueblank{私のかばんはあなたのよりずっと重い。}
        \vspace{0.5em}
        \item 2人のうち、彼女のほうが若い。
        \dialogueblank{She is the younger of the two.}
        \vspace{0.5em}
        \item Tokyo is far bigger than Sapporo.
        \dialogueblank{東京は札幌よりはるかに大きい。}
        \vspace{0.5em}
        \item この問題はあの問題より簡単だ。
        \dialogueblank{This problem is easier than that one.}
        \vspace{0.5em}
        \item Which is \underline{more popular}, soccer or baseball?
        \dialogueblank{サッカーと野球のどちらがより人気がありますか。}
        \vspace{0.5em}
        \item この2冊のうち、こちらのほうが新しい。
        \dialogueblank{This one is the newer of the two books.}
    \end{enumerate}
