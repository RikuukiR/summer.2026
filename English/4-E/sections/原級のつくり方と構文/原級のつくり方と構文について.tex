\subsection{原級のつくり方}
    \fitblankbf[]{as}と\fitblankbf[]{as}で原形(\fitblankbf[]{原級})をはさむだけである。

    否定形は\fitblankbf[]{not} \fitblankbf[]{as} \fitblankbf[]{原級} \fitblankbf[]{as} ～である。

\subsection{原級を使った構文(1)}
    \begin{theorembox}{\textbf{as＋原級＋as}}
        as ＋ 原級 ＋ as …

        \begin{flushright}
            『～と\fitblankbf[]{同じくらい}…』
        \end{flushright}
    \end{theorembox}
    \vspace{0.5em}

\subsection{原級を使った構文(2)}
    \begin{theorembox}{\textbf{not as＋原級＋as}}
        not as ＋ 原級 ＋ as …

        \begin{flushright}
            『～\fitblankbf[]{ほど}…\fitblankbf[]{でない}』
        \end{flushright}
    \end{theorembox}
    \vspace{0.5em}

\subsection{原級を使った構文(3)}
    \begin{theorembox}{\textbf{times as＋原級＋as}}
        times as ＋ 原級 ＋ as …

        \begin{flushright}
            『～の□\fitblankbf[]{倍}…』
        \end{flushright}
    \end{theorembox}
    \vspace{0.5em}

    ※『2倍』のときは\fitblankbf[]{twice}(＝ two times)。
    『半分、$\dfrac{1}{2}$』のときは\fitblankbf[]{half}が使われることもある。

\subsection{原級を使った構文(4)}
    \begin{theorembox}{\textbf{as＋原級＋as＋人＋can [ possible ]}}
        as ＋ 原級 ＋ as ＋ 人 ＋ (\fitblankbf[]{can})［ possible ］ …

        \begin{flushright}
            『\fitblankbf[]{できるだけ}～』
        \end{flushright}
    \end{theorembox}
    \vspace{0.5em}

\subsection{練習問題}
    \begin{enumerate}
        \item 彼は私と同じくらい速く泳げる。
        \dialogueblank{He can swim as fast as me.}
        \vspace{0.5em}
        \item This room is not as large as that one.
        \dialogueblank{この部屋はあの部屋ほど広くない。}
        \vspace{0.5em}
        \item 東京は札幌の2倍大きい。
        \dialogueblank{Tokyo is twice as large as Sapporo.}
        \vspace{0.5em}
        \item Please speak as loudly as you can.
        \dialogueblank{できるだけ大きな声で話してください。}
        \vspace{0.5em}
        \item My brother is as tall as my father.
        \dialogueblank{私の兄は父と同じくらい背が高い。}
        \vspace{0.5em}
        \item この問題はあの問題の3倍難しい。
        \dialogueblank{This problem is three times as difficult as that one.}
        \vspace{0.5em}
        \item She doesn't run as fast as her sister.
        \dialogueblank{彼女は姉ほど速く走らない。}
        \vspace{0.5em}
        \item Study as hard as possible.
        \dialogueblank{できるだけ一生懸命勉強しなさい。}
    \end{enumerate}
