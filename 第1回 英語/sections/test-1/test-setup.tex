\newcommand{\testblank@line}[2]{%
  \par\vspace{0.4em}%
  \noindent
  \ifshowanswer
    \underline{\makebox[#1][l]{{\color{blue}\textbf{#2}}}}%
  \else
    \underline{\makebox[#1][l]{\rule{0pt}{1.55\baselineskip}}}%
  \fi
}
\newcommand{\testblank}[2][\relax]{%
  \ifx#1\relax
    \testblank@line{#2}{}%
  \else
    \testblank@line{#2}{#1}%
  \fi
}

    \else
      \underline{\makebox[#2][l]{\rule{0pt}{1.35\baselineskip}}}%
    \fi
  \fi
}
