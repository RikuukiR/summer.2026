\subsection{一次関数について}

2つの数量 $x$，$y$ があり，$y$ が $x$ の一次式で表されるとき，

\[y = ax + b \qquad (a \neq 0)\]

という関係が成り立つ.

\begin{definitionbox}{\textbf{一次関数}}
    $y$ が $x$ の一次式 $y = ax + b$（$a \neq 0$）で表される関数を\textbf{一次関数}という.
    $a$ を\textbf{傾き}，$b$ を\textbf{切片}という.
\end{definitionbox}

\begin{definitionbox}{\textbf{正比例との関係}}
    一次関数 $y = ax + b$ において $b = 0$ のとき，$y = ax$ となり，これは\textbf{正比例}である.
\end{definitionbox}

\begin{theorembox}{\textbf{一次関数のグラフ}}
    $y = ax + b$ のグラフは\textbf{直線}である.
    \begin{enumerate}
        \item $a > 0$ のとき，グラフは\textbf{右上がり}である.
        \item $a < 0$ のとき，グラフは\textbf{右下がり}である.
        \item 切片は $b$ である（$x = 0$ のとき $y = b$）.
        \item $A \leqq x \leqq B$ のとき，$y$ の変域は $C \leqq y \leqq D$ となる.
    \end{enumerate}
\end{theorembox}

\begin{example}[【例題 1】]
    一次関数 $y = 2x - 1$ について，傾きと切片を求めなさい.
\end{example}

\begin{proof}\mbox{}\\
    $y = ax + b$ の形にそろえて読み取る.

    \begin{hiddenanswer}
        傾きは $a = 2$，切片は $b = -1$ である.
        \finalanswer{答え：傾き $2$，切片 $-1$}
    \end{hiddenanswer}
\end{proof}
