\subsection{反比例について}

2つの数量 $x$，$y$ があり，$y$ が $x$ に反比例するとき，

\[y = \frac{a}{x} \qquad (a \neq 0)\]

という関係が成り立つ. これは $xy = a$ と同じ意味を持つ.

\begin{definitionbox}{\textbf{反比例}}
    2つの数量 $x$，$y$ があり，$y$ が $x$ に反比例するとき，$y = \dfrac{a}{x}$（$a \neq 0$）という関係が成り立つ.
    このとき $a$ を\textbf{比例定数}という.
\end{definitionbox}

\begin{theorembox}{\textbf{反比例のグラフ}}
    $y = \dfrac{a}{x}$ のグラフは\textbf{双曲線}である.
    \begin{enumerate}
        \item $a > 0$ のとき，グラフは第1象限と第3象限にある.
        \item $a < 0$ のとき，グラフは第2象限と第4象限にある.
        \item $x$ の値が増えると $y$ の値は減る（符号に注意）.
    \end{enumerate}
\end{theorembox}

\begin{example}[【例題 1】]
    $y$ が $x$ に反比例し，$x = 4$ のとき $y = 3$ である. $y$ を $x$ の式で表しなさい.
\end{example}

\begin{proof}\mbox{}\\
    $y = \dfrac{a}{x}$ とおく.

    \begin{hiddenanswer}
        $x = 4$，$y = 3$ を代入すると，
        \begin{align*}
            3 &= \frac{a}{4} \\
            a &= 12
        \end{align*}
        \finalanswer{答え：$y = \dfrac{12}{x}$}
    \end{hiddenanswer}
\end{proof}
